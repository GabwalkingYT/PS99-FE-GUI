-- LocalScript inside StarterGui > ScreenGui
local Players = game:GetService("Players")
local ReplicatedStorage = game:GetService("ReplicatedStorage")
local RunService = game:GetService("RunService")

local player = Players.LocalPlayer
local ScreenGui = Instance.new("ScreenGui", player:WaitForChild("PlayerGui"))

ScreenGui.ResetOnSpawn = false

-- RemoteEvent for communication with the server
local spawnRemote = ReplicatedStorage:WaitForChild("SpawnRemote")

-- MAIN FRAME
local MainFrame = Instance.new("Frame", ScreenGui)
MainFrame.Name = "MainFrame"
MainFrame.Position = UDim2.new(0.3, 0, 0.3, 0)
MainFrame.Size = UDim2.new(0, 300, 0, 500)
MainFrame.BackgroundColor3 = Color3.fromRGB(255, 0, 0)
MainFrame.Active = true
MainFrame.Draggable = true
MainFrame.Visible = false -- Initially hidden

local Title = Instance.new("TextLabel", MainFrame)
Title.Size = UDim2.new(1, 0, 0, 40)

-- Function to create buttons for different spawners
local function makeButton(name, y, text)
    local btn = Instance.new("TextButton", MainFrame)
    btn.Name = name
    btn.Position = UDim2.new(0, 20, 0, y)
    btn.Size = UDim2.new(0, 260, 0, 40)
    btn.Text = text
    btn.BackgroundColor3 = Color3.fromRGB(50, 50, 50)
    btn.TextColor3 = Color3.new(1, 1, 1)
    btn.Font = Enum.Font.Gotham
    btn.TextSize = 14
    return btn
end

local PetSpawnerBtn = makeButton("PetSpawnerBtn", 60, "Pet Spawner")
local GemSpawnerBtn = makeButton("GemSpawnerBtn", 110, "Gem Spawner")
local HoverboardBtn = makeButton("HoverboardBtn", 160, "Hoverboard Spawner")
local GamepassBtn = makeButton("GamepassBtn", 210, "Gamepass Spawner")
local TitanicSpawnerBtn = makeButton("TitanicSpawnerBtn", 260, "Titanic Spawner")
local UltimateSpawnerBtn = makeButton("UltimateSpawnerBtn", 310, "Ultimate Spawner")

-- POPUP FRAME (Ultimate Spawner)
local Popup = Instance.new("Frame", ScreenGui)
Popup.Position = UDim2.new(0.35, 0, 0.35, 0)
Popup.Size = UDim2.new(0, 250, 0, 150)

local PopupTitle = Instance.new("TextLabel", Popup)

SpawnBtn.TextColor3 = Color3.new(1, 1, 1)

-- GUI logic
CloseBtn.MouseButton1Click:Connect(function()
    MainFrame.Visible = false
    OpenBtn.Visible = true
end)

OpenBtn.MouseButton1Click:Connect(function()
    MainFrame.Visible = true
    OpenBtn.Visible = false
end)

PetSpawnerBtn.MouseButton1Click:Connect(function() 
    Popup.Visible = true
    InputBox.Text = ""
end)

GemSpawnerBtn.MouseButton1Click:Connect(function() 
    Popup.Visible = true
    InputBox.Text = ""
end)

end)

    Popup.Visible = true

end)

UltimateSpawnerBtn.MouseButton1Click:Connect(function() 
    Popup.Visible = true
    InputBox.Text = ""
end)

-- Spawning logic for the Ultimate Spawner
SpawnBtn.MouseButton1Click:Connect(function()
    local amount = InputBox.Text  -- Accept both numbers and words
    if amount then
        Popup.Visible = false
        print("Spawning", amount)  -- Send spawn request to server via RemoteEvent
        spawnRemote:FireServer("ultimate", amount)  -- Replace with your actual spawning function
    end
end)

-- ðŸŒˆ Rainbow Background Animation
local hue = 0
RunService.RenderStepped:Connect(function()
    hue = (hue + 0.005) % 1
    local color = Color3.fromHSV(hue, 1, 1)
    MainFrame.BackgroundColor3 = color
    Popup.BackgroundColor3 = color
end)
