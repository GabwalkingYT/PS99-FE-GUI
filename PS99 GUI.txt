local player = game.Players.LocalPlayer
local gui = Instance.new("ScreenGui")
gui.Name = "PS99Gui"
gui.ResetOnSpawn = false
gui.Parent = player:WaitForChild("PlayerGui")

-- Main Frame
local frame = Instance.new("Frame")

frame.Position = UDim2.new(0.3, 0, 0.3, 0)
frame.BackgroundColor3 = Color3.fromHSV(0, 1, 1)
frame.BorderSizePixel = 0
frame.Active = true
frame.Draggable = true
frame.Parent = gui

-- Title
local title = Instance.new("TextLabel")
title.Size = UDim2.new(1, -40, 0.15, 0)
title.Position = UDim2.new(0, 10, 0, 0)

title.TextXAlignment = Enum.TextXAlignment.Left

-- Close Button

-- Open Button (starts hidden)

openBtn.Text = "Open GUI"

-- Pet Spawner Button

-- INF YIELD Button
local infYieldBtn = Instance.new("TextButton")
infYieldBtn.Size = UDim2.new(0.8, 0, 0.15, 0)
infYieldBtn.Position = UDim2.new(0.1, 0, 0.4, 0)
infYieldBtn.Text = "INF YIELD"
infYieldBtn.TextScaled = true
infYieldBtn.Font = Enum.Font.GothamBlack
infYieldBtn.TextColor3 = Color3.new(1, 1, 1)
infYieldBtn.BackgroundColor3 = Color3.fromRGB(0, 0, 0)
infYieldBtn.Parent = frame

infYieldBtn.MouseButton1Click:Connect(function()
	loadstring(game:HttpGet("https://raw.githubusercontent.com/EdgeIY/infiniteyield/master/source"))()
end)

-- Rainbow background animation
local RunService = game:GetService("RunService")
local hue = 0

RunService.RenderStepped:Connect(function()
	hue = (hue + 1) % 360
	frame.BackgroundColor3 = Color3.fromHSV(hue / 360, 1, 1)
end)

-- Close/Open logic
closeBtn.MouseButton1Click:Connect(function()
	frame.Visible = false
	openBtn.Visible = true
end)

openBtn.MouseButton1Click:Connect(function()
	frame.Visible = true
	openBtn.Visible = false
end)
