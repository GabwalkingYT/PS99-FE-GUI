-- Show "Spawned!!" Popup
local function showSpawnedPopup()
	SpawnedPopup.Visible = true
	wait(2)  -- Show for 2 seconds
	SpawnedPopup.Visible = false
end

-- Show popup function
local function showPopup(spawnerType)
	currentSpawner = spawnerType
	Popup.Visible = true
	InputBox.Text = ""
end
